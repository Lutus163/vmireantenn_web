git init - создать локальный репозиторий

git remote add origin https://github.com/Lutus163/telegram_bot.git - связать репозитории

git clone https://github.com/Lutus163/telegram_bot.git - клонирование репозитория
git branch -r - посмотреть все ветки
git fetch --all - получить все ветки
git origin название ветки - переключиться на ветку

git status - все не сохраненные версии файлов

git add "Bot.py" или . для всех файлов из локального репозитория

git commit -m "updated README" - создание чекпоинта файла

git log - все коммиты

git checkout ... - вернуться к версии файла через хеш

git push имя репозитория обычно это origin master - загрузить на гит в ветку свои изменения

git pull origin master - скачать все новые файлы в локальный репозиторий из ветки

git branch - посмотреть все ветки

git branch имя - создать ветку

git checkout имя ветки - переключиться на ветку

git merge dev - провести слияние веток

git rebase dev - слияние веток вместе с комментариями
